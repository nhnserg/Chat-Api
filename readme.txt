# Подключение к WebSocket серверу
wscat -c ws://localhost:3000

# Отправка сообщения для подключения к комнате
# Замените room_id на реальный идентификатор комнаты и TOKEN на реальный JWT токен
{
  "type": "join",
  "roomId": "room_id",  # Идентификатор комнаты
  "token": "TOKEN"      # Ваш JWT токен
}

# Отправка сообщения в комнату
# Замените room_id на реальный идентификатор комнаты и TOKEN на реальный JWT токен
{
  "type": "message",
  "roomId": "room_id",  # Идентификатор комнаты
  "content": "Привет!",  # Содержимое сообщения
  "token": "TOKEN"      # Ваш JWT токен
}

# Отправка личного сообщения
# Замените username на имя получателя и TOKEN на реальный JWT токен
{
  "type": "privateMessage",
  "recipient": "username",  # Имя пользователя-получателя
  "content": "Привет!",      # Содержимое сообщения
  "token": "TOKEN"          # Ваш JWT токен
}

# Подключение нескольких экземпляров для тестирования:
# Откройте несколько окон с `wscat` и повторите все вышеуказанные запросы
# Это поможет протестировать:
# - Подключение нескольких пользователей к одной комнате
# - Отправку сообщений в чат
# - Отправку личных сообщений между пользователями
# - Статус набора текста
# - Отключение пользователей (например, через "type": "leave")
